## <a id="importing"></a>Importing and usage

```jsx
...
import SnackBar from '@catho/quantum/SnackBar';

export default function Template () {
  const [openSnackBar, setOpenSnackbar] = useState(false);
  return (
    <>
      <Button onClick={() => setOpenSnackbar(true)}>Open snackbar</Button>
      {openSnackBar && <SnackBar onClose={() => setOpenSnackbar(false)} />}
    </>
  );
};
```

### With action trigger

```jsx
...
import SnackBar from '@catho/quantum/SnackBar';

export default function Template () {
  const [openSnackBar, setOpenSnackbar] = useState(false);
  return (
    <>
      <Button onClick={() => setOpenSnackbar(true)}>Open snackbar</Button>
      {openSnackBar && <SnackBar actionTrigger={{ title: 'HIDE', callbackFn: () => setOpenSnackbar(false)}} onClose={() => setOpenSnackbar(false)} />}
    </>
  );
};
```
