## <a id="importing"></a>Importing and usage

```jsx
...
import Dialog from '@catho/quantum/Dialog';
import Button from '@catho/quantum/Button';

export default function SimpleDialog() {
  const [openDialog, setOpenDialog] = useState(false);

  return (
    <>
      <Button onClick={() => setOpenDialog(true)}>Click to open dialog</Button>
      {openDialog && <Dialog>Content...</Dialog>}
    </>
  );
}
```

### Close On Time

```jsx
...
import Dialog from '@catho/quantum/Dialog';
import Button from '@catho/quantum/Button';

export default function SimpleDialog() {
  const [openDialog, setOpenDialog] = useState(false);

  return (
    <>
      <Button onClick={() => setOpenDialog(true)}>Click to open dialog</Button>
      {openDialog && <Dialog closeOnTime={3} >Content...</Dialog>}
    </>
  );
}
```
